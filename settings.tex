\newcommand{\SectionDesign}[4]{
    \ifthenelse{\equal{#1}{section}}{\vspace{0cm}}{\vspace{0.5cm}}
    \noindent
    \csname #1*\endcsname{\textcolor[HTML]{1E90FF}{\fontsize{#2pt}{#3pt}\selectfont #4}}
    \ifthenelse{\equal{#1}{section}}{\vspace{2cm}}{\vspace{0.5cm}}
}

\newcommand{\TextAndImage}[7][{}]{
    \fontsize{#6pt}{#7pt}\selectfont
    \noindent
    \begin{minipage}[c]{#4\textwidth}
    \RaggedRight
    #2 % First parameter: text
    \end{minipage}
    \hfill
    \begin{minipage}[c]{#5\textwidth}
    \includegraphics[width=\textwidth, #1]{#3} % Second parameter: image file name
    \end{minipage}
}

\newcommand{\ImageAndText}[7][{}]{
    \fontsize{#6pt}{#7pt}\selectfont

    \begin{minipage}[c]{#4\textwidth}
    \includegraphics[width=\textwidth, #1]{#2} % Second parameter: image file name
    \end{minipage}
    \hfill
    \begin{minipage}[c]{#5\textwidth}
    #3 % First parameter: text
    \end{minipage}
}
